nom_usuario varchar (40) primary key,
contra varchar (60) not null
);

create table estanque(
nombre varchar (30) primary key,
tipo_estanque varchar (30) not null,
area_metros int not null,
estado varchar (30) not null
);

create table alimentacion(
fecha date,
hora time,
purina varchar (20) not null,
lote_alim varchar (10) not null,
cantidad int not null,
estanque varchar (25) not null,
operario varchar (40) not null
);
